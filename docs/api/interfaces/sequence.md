```python
        return False
    except SequenceExecutionError as e:
        logger.error(f"–û—à–∏–±–∫–∞ –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è: {e}")
        return False
    except Exception as e:
        logger.error(f"–ù–µ–æ–∂–∏–¥–∞–Ω–Ω–∞—è –æ—à–∏–±–∫–∞: {e}")
        return False

def _execute_with_timeout(self, sequence_name: str) -> bool:
    """–í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ —Å —Ç–∞–π–º–∞—É—Ç–æ–º"""
    import threading
    import time
    
    sequence = self._sequences[sequence_name]
    timeout = sequence.get('timeout', 300.0)
    
    # –°–æ–∑–¥–∞–Ω–∏–µ —Å–æ–±—ã—Ç–∏—è –¥–ª—è –æ—Å—Ç–∞–Ω–æ–≤–∫–∏
    stop_event = threading.Event()
    
    def execution_thread():
        try:
            self._running_sequences.add(sequence_name)
            result = self._execute_sequence_internal(sequence_name, stop_event)
            return result
        finally:
            self._running_sequences.discard(sequence_name)
    
    # –ó–∞–ø—É—Å–∫ –≤ –æ—Ç–¥–µ–ª—å–Ω–æ–º –ø–æ—Ç–æ–∫–µ
    thread = threading.Thread(target=execution_thread)
    thread.daemon = True
    thread.start()
    
    # –û–∂–∏–¥–∞–Ω–∏–µ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –∏–ª–∏ —Ç–∞–π–º–∞—É—Ç–∞
    thread.join(timeout=timeout)
    
    if thread.is_alive():
        stop_event.set()
        thread.join(timeout=5.0)  # –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω–æ–µ –≤—Ä–µ–º—è –¥–ª—è –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ–≥–æ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è
        logger.error(f"–ü—Ä–µ–≤—ã—à–µ–Ω —Ç–∞–π–º–∞—É—Ç –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏: {sequence_name}")
        return False
    
    return True
```
